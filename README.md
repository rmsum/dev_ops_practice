# linux_commands for file handling

# Create a directory called "my_folder"
$ mkdir my_folder

# navigate into it
$ cd my_folder

# create a file named "my_file.txt" with some text using vim editer. 
$ vi my_file.txt

# create another file named ""another_file.txt"" with some text.
$ echo "This is a content of another file.txt" > another_file.txt

# Concatenate the content of ""another_file.txt"" to ""my_file.txt"" 
$ cat another_file.txt > my_file.txt

# display the updated content. 
$ cat my_file.txt

# list all files and directories in the current directory.
$ ls
